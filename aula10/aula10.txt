show databases;

-- para ativar um esquema ou um banco de dados
use mysql;

-- mostra as tabelas de um esquema ou banco de dados
show tables;

-- mostra todas as colunas/atributos e todas as linhas da tabela user
SELECT *
FROM user;

-- crie um esquema de banco de dados
CREATE SCHEMA ufn_db; 

-- usa o banco criado
use ufn_db;
show tables;

-- cria a tabela curso (id_curso, descricao)

CREATE TABLE curso (
	id_curso INT PRIMARY KEY NOT NULL,
    descricao VARCHAR(50) NOT NULL
);

-- cria tabela aluno

CREATE TABLE aluno (
	id_aluno INT PRIMARY KEY NOT NULL,
    nome VARCHAR(50) NOT NULL,
    id_curso INT,
    CONSTRAINT fk_curso FOREIGN KEY (id_curso) REFERENCES curso(id_curso)
);
